import tkinter as tk from tkinter import messagebox import math def calculate_probability(): try: UK = float(entry_UK.get()) SF36_PCS = float(entry_SF36_PCS.get()) SF36_MCS = float(entry_SF36_MCS.get()) KPS = float(entry_KPS.get()) exponent = 4.9654 + UK + SF36_PCS + SF36_MCS + KPS P = math.exp(exponent) / (1 + math.exp(exponent)) result_var.set(f"Probability P = {P:.4f}") except ValueError: messagebox.showerror("Input error", "Please enter valid numerical values.") # Create main window root = tk.Tk() root.title("Probability Calculator") # Labels and entries tk.Label(root, text="UK:").grid(row=0, column=0, padx=10, pady=5) entry_UK = tk.Entry(root) entry_UK.grid(row=0, column=1, padx=10, pady=5) tk.Label(root, text="SF36-PCS:").grid(row=1, column=0, padx=10, pady=5) entry_SF36_PCS = tk.Entry(root) entry_SF36_PCS.grid(row=1, column=1, padx=10, pady=5) tk.Label(root, text="SF36-MCS:").grid(row=2, column=0, padx=10, pady=5) entry_SF36_MCS = tk.Entry(root) entry_SF36_MCS.grid(row=2, column=1, padx=10, pady=5) tk.Label(root, text="KPS:").grid(row=3, column=0, padx=10, pady=5) entry_KPS = tk.Entry(root) entry_KPS.grid(row=3, column=1, padx=10, pady=5) # Button to calculate calc_button = tk.Button(root, text="Calculate Probability", command=calculate_probability) calc_button.grid(row=4, column=0, columnspan=2, pady=10) # Label to display result result_var = tk.StringVar() result_label = tk.Label(root, textvariable=result_var, font=("Arial", 14)) result_label.grid(row=5, column=0, columnspan=2, pady=5) root.mainloop()
